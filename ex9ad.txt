package com.example.exp9emo;

import android.graphics.Color;
import android.os.Bundle;
import android.view.ContextMenu;
import android.view.MenuItem;
import android.view.View;
import android.widget.RelativeLayout;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

import java.nio.file.Path;

public class MainActivity extends AppCompatActivity {
    TextView textView;
    RelativeLayout relativeLayout;
    private Path Log;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);


        textView = (TextView) findViewById(R.id.textView);
        relativeLayout = (RelativeLayout) findViewById(R.id.relLayout);


        registerForContextMenu(textView);

    }

    @Override
    public void onCreateContextMenu(ContextMenu menu, View v, ContextMenu.ContextMenuInfo menuInfo) {
        super.onCreateContextMenu(menu, v, menuInfo);

        menu.setHeaderTitle("Choose a app");

        menu.add(0, v.getId(), 0, "Alarm");
        menu.add(0, v.getId(), 0, "Social Media");
        menu.add(0, v.getId(), 0, "Instagram");
        menu.add(0, v.getId(), 0, "Twitter");
    }

    // menu item select listener
    @Override
    public boolean onContextItemSelected(MenuItem item) {
        if (item.getTitle() == "Alarm") {
            relativeLayout.setBackgroundColor(Color.GRAY);
            textView.setText("alarm is selected");
        } else if (item.getTitle() == "Social Media") {
            relativeLayout.setBackgroundColor(Color.GRAY);
            textView.setText("Social Media is selected");
        } else if (item.getTitle() == "Instagram") {
            relativeLayout.setBackgroundColor(Color.GRAY);
            textView.setText("Instagram is selected");
        }
        else if (item.getTitle() == "Twitter") {
            relativeLayout.setBackgroundColor(Color.GRAY);
            textView.setText("Twitter is selected");
        }
        return true;
    }
}
